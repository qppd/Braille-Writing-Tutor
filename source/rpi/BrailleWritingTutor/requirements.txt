## Requirements for Braille Writing Tutor (Raspberry Pi)

# GPIO control for Raspberry Pi
pigpio==1.78

# Text-to-Speech library
gtts==2.5.4

# Audio processing for TTS playback
pygame==2.6.1

# Serial communication with Arduino
pyserial==3.5

# Development and debugging
pytest==8.4.2

## Installation Instructions:
# 1. Update system packages:
#    sudo apt update && sudo apt upgrade -y
#
# 2. Install system dependencies:
#    sudo apt install python3-pip python3-dev pigpio python3-pigpio
#    sudo apt install espeak espeak-data
#
# 3. Enable and start pigpiod daemon:
#    sudo systemctl enable pigpiod
#    sudo systemctl start pigpiod
#
# 4. Install Python packages:
#    pip3 install -r requirements.txt
#
# 5. Enable GPIO access (if needed):
#    sudo usermod -a -G gpio $USER
#    # Logout and login again
#
# 6. Test pigpio connection:
#    python3 -c "import pigpio; pi = pigpio.pi(); print('Connected!' if pi.connected else 'Failed'); pi.stop()"
#
# OR use the automated setup script:
#    chmod +x setup_pigpio.sh
#    ./setup_pigpio.sh

## Notes:
# - pigpio provides better GPIO control than RPi.GPIO
# - No sudo required to run the application
# - Make sure pigpiod daemon is running before starting the app
# - For development/testing on non-RPi systems, consider using:
#   - Mock objects for testing (see test_components.py)
